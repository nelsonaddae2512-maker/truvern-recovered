// app/(app)/vendors/[id]/assessments/manage/page.tsx
import Link from "next/link";
import { redirect, notFound } from "next/navigation";
import prisma from "@/lib/prisma";
import { requireDbOrganization } from "@/lib/org-db";

import UnapplyFrameworkButton from "./unapply-framework-button.client";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";
export const revalidate = 0;

type SearchParams = Record<string, string | string[] | undefined>;
type Props = { params: { id: string }; searchParams?: SearchParams | Promise<SearchParams> };

function pickOne(v?: string | string[]) {
  return Array.isArray(v) ? v[0] : v;
}
function safeStr(v: unknown) {
  return typeof v === "string" ? v.trim() : "";
}
function num(v: unknown): number | null {
  const n = Number(String(v ?? "").trim());
  return Number.isFinite(n) && n > 0 ? n : null;
}
function clsx(...parts: Array<string | false | null | undefined>) {
  return parts.filter(Boolean).join(" ");
}
function fmtDate(d: unknown) {
  const dt = d instanceof Date ? d : new Date(String(d));
  if (!Number.isFinite(dt.getTime())) return "—";
  return dt.toLocaleDateString(undefined, { year: "numeric", month: "short", day: "numeric" });
}

async function resolveSearchParams(sp?: SearchParams | Promise<SearchParams>): Promise<SearchParams> {
  // Next.js sometimes passes searchParams as a Promise; handle both without `any`.
  const maybePromise = sp as unknown;
  if (
    !!maybePromise &&
    (typeof maybePromise === "object" || typeof maybePromise === "function") &&
    "then" in (maybePromise as Record<string, unknown>) &&
    typeof (maybePromise as Record<string, unknown>).then === "function"
  ) {
    return (await (sp as Promise<SearchParams>)) ?? {};
  }
  return (sp as SearchParams) ?? {};
}

export default async function ManageFrameworkPage({ params, searchParams }: Props) {
  const org = await requireDbOrganization().catch(() => null);
  if (!org) redirect("/");

  const vendorId = num(params?.id);
  if (!vendorId) return notFound();

  const sp = await resolveSearchParams(searchParams);

  const templateId = num(pickOne(sp?.templateId));
  if (!templateId) redirect(`/vendors/${vendorId}/assessments`);

  const vendor = await prisma.vendor.findFirst({
    where: { id: vendorId, organizationId: org.id },
    select: { id: true, name: true },
  });
  if (!vendor) return notFound();

  const template = await prisma.assessmentTemplate.findFirst({
    where: { id: templateId, organizationId: org.id },
    select: {
      id: true,
      name: true,
      description: true,
      standard: true,
      version: true,
      category: true,
    },
  });

  const runs = await prisma.assessmentRun.findMany({
    where: { organizationId: org.id, vendorId, templateId },
    orderBy: [{ updatedAt: "desc" }, { createdAt: "desc" }],
    take: 50,
    select: {
      id: true,
      status: true,
      createdAt: true,
      updatedAt: true,
    },
  });

  const hubHref = `/vendors/${vendorId}/assessments`;

  return (
    <div className="mx-auto max-w-6xl px-6 pb-16 pt-10">
      <div className="rounded-2xl border border-white/10 bg-white/[0.06] p-6 shadow-[0_0_0_1px_rgba(34,211,238,0.08)] backdrop-blur-xl">
        <div className="flex flex-wrap items-start justify-between gap-3">
          <div>
            <div className="text-xs font-semibold tracking-wide text-cyan-200/80">Manage framework</div>
            <div className="mt-1 text-2xl font-semibold text-white">
              {safeStr(template?.name) || `Template #${templateId}`}
            </div>
            <div className="mt-1 text-sm text-white/60">Vendor: {vendor.name}</div>
            {!!safeStr(template?.description) && (
              <div className="mt-2 text-sm text-white/55">{template?.description}</div>
            )}
          </div>

          <div className="flex flex-wrap items-center gap-2">
            <Link
              href={hubHref}
              prefetch={false}
              className="rounded-full border border-white/10 bg-white/[0.06] px-3 py-1.5 text-xs font-semibold text-white/85 hover:bg-white/[0.09]"
            >
              ← Back to hub
            </Link>

            <Link
              href="/assessment-templates/catalog"
              prefetch={false}
              className="rounded-full border border-cyan-300/20 bg-cyan-400/10 px-3 py-1.5 text-xs font-semibold text-cyan-100 hover:bg-cyan-400/15"
            >
              Template Catalog →
            </Link>

            <UnapplyFrameworkButton
              vendorId={vendorId}
              templateId={templateId}
              templateName={template?.name}
              backHref={hubHref}
            />
          </div>
        </div>
      </div>

      <div className="mt-8">
        <div className="text-sm font-semibold text-white/85">Runs</div>
        <div className="mt-1 text-sm text-white/50">Open a run to answer questions and compute scoring.</div>

        <div className="mt-4 space-y-3">
          {runs.length === 0 ? (
            <div className="rounded-2xl border border-white/10 bg-white/[0.04] p-5 text-sm text-white/60">
              No runs found for this framework.
            </div>
          ) : (
            runs.map((r) => (
              <div
                key={r.id}
                className="flex flex-wrap items-center justify-between gap-3 rounded-2xl border border-white/10 bg-white/[0.04] p-5"
              >
                <div>
                  <div className="flex flex-wrap items-center gap-2">
                    <div className="text-sm font-semibold text-white">Run #{r.id}</div>
                    <span className="rounded-full border border-white/10 bg-white/[0.06] px-2.5 py-0.5 text-xs font-semibold text-white/70">
                      {String(r.status || "—")}
                    </span>
                    <span className="text-xs text-white/45">Updated: {fmtDate(r.updatedAt)}</span>
                  </div>
                  <div className="mt-1 text-xs text-white/40">
                    Created: {fmtDate(r.createdAt)} · Updated: {fmtDate(r.updatedAt)}
                  </div>
                </div>

                <Link
                  href={`/assessment/runs/${r.id}`}
                  prefetch={false}
                  className={clsx(
                    "inline-flex items-center justify-center rounded-xl px-4 py-2.5 text-sm font-semibold",
                    "border border-cyan-300/20 bg-cyan-400/10 text-cyan-50 hover:bg-cyan-400/15"
                  )}
                >
                  Open workspace →
                </Link>
              </div>
            ))
          )}
        </div>
      </div>
    </div>
  );
}
